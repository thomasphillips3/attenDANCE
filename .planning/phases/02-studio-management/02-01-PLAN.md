    Create supabase/migrations/20260522000600_enrollment_functions.sql containing three functions and one trigger:

    1. CREATE OR REPLACE FUNCTION enroll_student(p_organization_id uuid, p_student_id uuid, p_class_id uuid) RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER:
    - DECLARE v_capacity smallint, v_active_count bigint, v_status enrollment_status, v_enrollment_id uuid, v_existing record
    - Lock the class row: SELECT capacity INTO v_capacity FROM classes WHERE id = p_class_id AND organization_id = p_organization_id FOR UPDATE. If NOT FOUND, return jsonb_build_object('error', 'Class not found').
    - Check existing enrollment: SELECT id, status INTO v_existing FROM enrollments WHERE student_id = p_student_id AND class_id = p_class_id AND organization_id = p_organization_id. If found and status = 'active', return error 'Student already enrolled'. If found and status = 'waitlist', return error 'Student already on waitlist'.
    - Count active: SELECT COUNT(*) INTO v_active_count FROM enrollments WHERE class_id = p_class_id AND organization_id = p_organization_id AND status = 'active'
    - Set status: IF v_capacity IS NULL OR v_active_count < v_capacity THEN v_status := 'active' ELSE v_status := 'waitlist'
    - Upsert: INSERT INTO enrollments (organization_id, student_id, class_id, status) VALUES (...) ON CONFLICT (student_id, class_id) DO UPDATE SET status = v_status, enrolled_at = now(), dropped_at = NULL, updated_at = now() RETURNING id INTO v_enrollment_id
    - Return jsonb_build_object('enrollmentId', v_enrollment_id, 'status', v_status::text, 'activeCount', v_active_count + CASE WHEN v_status = 'active' THEN 1 ELSE 0 END, 'capacity', v_capacity)

    2. CREATE OR REPLACE FUNCTION promote_from_waitlist() RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER:
    - Only fire when status changes TO 'dropped': IF NEW.status != 'dropped' OR OLD.status = 'dropped' THEN RETURN NEW
    - SELECT capacity INTO v_capacity FROM classes WHERE id = NEW.class_id FOR UPDATE
    - SELECT COUNT(*) INTO v_active_count FROM enrollments WHERE class_id = NEW.class_id AND organization_id = NEW.organization_id AND status = 'active'
    - IF v_capacity IS NULL OR v_active_count < v_capacity THEN find earliest waitlisted: SELECT id INTO v_waitlisted_id FROM enrollments WHERE class_id = NEW.class_id AND organization_id = NEW.organization_id AND status = 'waitlist' ORDER BY enrolled_at ASC LIMIT 1 FOR UPDATE SKIP LOCKED. If found, UPDATE enrollments SET status = 'active', updated_at = now() WHERE id = v_waitlisted_id.
    - RETURN NEW

    3. CREATE TRIGGER trg_promote_from_waitlist AFTER UPDATE ON enrollments FOR EACH ROW EXECUTE FUNCTION promote_from_waitlist()

    4. CREATE OR REPLACE FUNCTION transfer_student(p_organization_id uuid, p_student_id uuid, p_from_class_id uuid, p_to_class_id uuid) RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER:
    - Per user decision: atomic single transaction
    - UPDATE enrollments SET status = 'dropped', dropped_at = now(), updated_at = now() WHERE student_id = p_student_id AND class_id = p_from_class_id AND organization_id = p_organization_id AND status IN ('active', 'waitlist'). Check ROW_COUNT; if 0, return error 'Student not enrolled in source class'.
    - Call v_enroll_result := enroll_student(p_organization_id, p_student_id, p_to_class_id)
    - If v_enroll_result ? 'error', return jsonb_build_object('error', v_enroll_result->>'error', 'note', 'Student was dropped from source class but could not enroll in target')
    - Return jsonb_build_object('fromClassId', p_from_class_id, 'fromStatus', 'dropped', 'toClassId', p_to_class_id, 'toStatus', v_enroll_result->>'status', 'enrollmentId', v_enroll_result->>'enrollmentId')

    5. GRANT EXECUTE on all three functions to authenticated and service_role roles.

    Create supabase/migrations/20260522000700_storage_bucket.sql:
    - Per user decision: PRIVATE bucket (public=false) with signed URLs
    - CHILDREN'S DATA: Photos are of minors. Storage policies must be organization-scoped, not just authenticated.
    - Upload path convention: {organization_id}/{student_id}/{timestamp}.jpg — embeds org context in the path.
    - INSERT INTO storage.buckets (id, name, public, file_size_limit, allowed_mime_types) VALUES ('student-photos', 'student-photos', false, 2097152, ARRAY['image/jpeg','image/png','image/webp'])
    - CREATE POLICY "Staff can upload student photos to their org" ON storage.objects FOR INSERT TO authenticated WITH CHECK (bucket_id = 'student-photos' AND (storage.foldername(name))[1] = (auth.jwt() -> 'app_metadata' ->> 'organization_id'))

    - CREATE POLICY "Staff can delete student photos in their org" ON storage.objects FOR DELETE TO authenticated USING (bucket_id = 'student-photos' AND (storage.foldername(name))[1] = (auth.jwt() -> 'app_metadata' ->> 'organization_id'))
    - This ensures when parents get accounts in Phase 4, they cannot access photos from other organizations.
